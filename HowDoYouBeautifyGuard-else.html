Greg Titus 认为使用一个简单的语句和一个不会返回的函数会更好:
viagreg.swift hosted with ❤ by GitHub

// Usage
func usage() -> Never {
    print("Usage: \(appName) ")
    exit(-1)
}
guard !arguments.isEmpty,
    nums.count == arguments.count else { usage() }
